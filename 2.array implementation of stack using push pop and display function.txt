
#include <stdio.h>
#define MAX 10
int st[MAX], top=-1;
void push();
void pop();
void peek();
void display();
int main()
{
int val, ch;
printf("*********Stack operations
using array*********");
printf("\n *****MAIN MENU*****");
printf("\n 1. PUSH \n 2. POP \n 3.
Peek \n 4. display \n 5. Exit");
do
{
printf("\n Please Enter your Choice:
");
scanf("%d", &ch);
switch(ch)
{
case 1:
push();
break;
case 2:
pop();
break;
 case 3:
peek();
break;
case 4:
display();
 break;

 case 5:
printf("Exiting....");
break;
default:
printf("Please Enter valid choice ");
}
}while(ch != 5);
return 0;
}
void push()
{
int val;
printf("\n Enter the number to be
pushed on stack: ");
scanf("%d", &val);
if(top == MAX-1)
{
printf("\n STACK OVERFLOW");
}
 else
 {
 top++;
 st[top] = val;
printf("%d Value PUSHED
Successfully\n", val);
}
}
void pop()
{
int val;
if(top == -1)
{
printf("\n STACK UNDERFLOW");
}
 else
{
val= st[top];
top--;

printf("%d Value POPED Successfully\n",
val);
}
}
 void peek()
{
if (top == -1)
 {
 printf("Underflow");
}
else
{
 printf("\n The value stored at top of
stack is: %d\n", st[top]);
}
}
void display()
{
 int i;
 if(top == -1)
printf("\n STACK IS EMPTY");
else
{
for(i=top;i>=0;i--)
printf("\n %d",st[i]);
printf("\n");
}
}
