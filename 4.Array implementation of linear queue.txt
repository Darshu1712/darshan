#include <stdio.h>
#define MAX 50
void addqueue(int);
int dequeue();
int A[MAX];
int Rear=-1;
int Front=-1;
int main()
{
    int num,choice,i;
    do
    {
        printf("\n1.add \n2.dalete \n3.display \n4.exit");
        printf("\n enter option:");
        scanf("%d",&choice);
        switch(choice)
        {
            case 1:printf("enter no to add in queue:");
            scanf("%d",&num);
            addqueue(num);
            break;
            case 2:num=dequeue();
            printf("deleted elements=%d",num);
            break;
            case 3:for(i=Front+1;i<=Rear;i++)
            printf("\t%d",A[i]);
            break;
            case 4:
            default:printf("\n invalid option");
            break;
        }
    }
while(choice!=4);
return 0;
}
void addqueue(int num)
{
    if(Rear==MAX-1)
    {
        printf("\nQueue FULL");
    }
    else
    {
        Rear++;
        A[Rear]=num;
    }
}
int dequeue()
{
    int n;
    if(Front==-1&&Rear==-1)
    {printf("\n queue in empty");
    return(0);
}
else
{
    Front++;
    n=A[Front];
    return(n);
}
}
